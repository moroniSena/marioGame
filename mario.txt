        private void timer1_Tick(object sender, EventArgs e)
        {
            SideCollision();

            index++;

            points -= one;

            score2.Text = Convert.ToString(savedScore);

            score.Text = Convert.ToString(points);

            if (left2 == true)
            {
                goomba.Left -= 3;
            }

            if (right2 == true)
            {
                goomba.Left += 3;
            }
            if (left4 == true)
            {
                goomba2.Left -= 3;
            }

            if (right4 == true)
            {
                goomba2.Left += 3;
            }
            if (left3 == true)
            {
                koopa.Left -= 3;
            }

            if (right3 == true)
            {
                koopa.Left += 3;
            }
            if (right == true && index % 15 == 0)
            {
                player.Image = Image.FromFile("Mario_walk_smb.gif");
            }

            if (left == true && index % 15 == 0)
            {
                player.Image = Image.FromFile("walkL.gif");
            }

            if (right == true)
            {
                player.Left += 5;
            }
            if (left == true)
            {
                player.Left -= 5;
            }
            if (jump == true)
            {
                player.Top -= force;
                force --;
            }
            if (player.Top + player.Height >= screen.Height)
            {
                player.Top = screen.Height - player.Height;

                if (jump == true)
                {
                    player.Image = Image.FromFile("pixel-mario.png");
                }

                jump = false;
            }
            else
            {
                player.Top += power;
            }

            HeadCollision();

            TopCollision();
            
        }


        private void Form1_KeyUp(object sender, KeyEventArgs e)
        {
            if (e.KeyCode == Keys.Right)
            {
                right = false;
                
                    player.Image = Image.FromFile("pixel-mario.png");
               
            }
            if (e.KeyCode == Keys.Left)
            {
                left = false;
                
                    player.Image = Image.FromFile("pixel-marioL.png");
               
            }
        }
